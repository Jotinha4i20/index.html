    <script>
        let totalBruto = 0;
        let totalLiquido = 0;

        const form = document.getElementById('registroForm');
        const tabela = document.getElementById('tabelaRegistros').getElementsByTagName('tbody')[0];
        const displayTotalBruto = document.getElementById('totalBruto');
        const displayTotalLiquido = document.getElementById('totalLiquido');

        form.addEventListener('submit', function(e) {
            e.preventDefault();

            // Captura os valores
            const cliente = document.getElementById('cliente').value;
            const tipoContrato = document.getElementById('tipoContrato').value;
            const valorBruto = parseFloat(document.getElementById('valorBruto').value);
            const valorLiquido = parseFloat(document.getElementById('valorLiquido').value);

            // Atualiza os painéis superiores
            totalBruto += valorBruto;
            totalLiquido += valorLiquido;
            
            const formatadorMeda = new Intl.NumberFormat('pt-BR', { style: 'currency', currency: 'BRL' });
            
            displayTotalBruto.innerText = formatadorMeda.format(totalBruto);
            displayTotalLiquido.innerText = formatadorMeda.format(totalLiquido);

            // Insere os dados na tabela
            const novaLinha = tabela.insertRow();
            novaLinha.insertCell(0).innerText = cliente;
            novaLinha.insertCell(1).innerText = tipoContrato;
            novaLinha.insertCell(2).innerText = formatadorMeda.format(valorBruto);
            novaLinha.insertCell(3).innerText = formatadorMeda.format(valorLiquido);

            // Limpa os campos para o próximo registro
            form.reset();
        });
    </script>
